JavaScript 程式碼邏輯參考

JavaScript
// 初始化屬性
let player = {
    x: 100, y: 100,
    vx: 0, vy: 0,        // 速度 (Velocity)
    accel: 0.5,         // 加速度
    friction: 0.95,      // 穩定器開啟時的減速係數
    isDampenerOn: true   // 穩定器開關
};

function update() {
    // 1. 處理輸入 (加速)
    if (keys['ArrowUp'])    player.vy -= player.accel;
    if (keys['ArrowDown'])  player.vy += player.accel;
    if (keys['ArrowLeft'])  player.vx -= player.accel;
    if (keys['ArrowRight']) player.vx += player.accel;

    // 2. 處理慣性與穩定器邏輯
    if (player.isDampenerOn) {
        // 如果沒有按下任何移動鍵，就自動減速
        if (!keys['ArrowUp'] && !keys['ArrowDown']) player.vy *= player.friction;
        if (!keys['ArrowLeft'] && !keys['ArrowRight']) player.vx *= player.friction;
        
        // 當速度極小時直接歸零，避免無限小數運算
        if (Math.abs(player.vx) < 0.1) player.vx = 0;
        if (Math.abs(player.vy) < 0.1) player.vy = 0;
    }

    // 3. 更新位置
    player.x += player.vx;
    player.y += player.vy;
}

JavaScript
// 核心邏輯預覽
class Astronaut {
    constructor() {
        this.x = 400;
        this.y = 300;
        this.vx = 0; // 水平速度
        this.vy = 0; // 垂直速度
        this.accel = 0.2; // 噴氣加速度
        this.friction = 0.96; // 穩定器摩擦力
        this.dampener = true; // 穩定器開關
    }

    update(keys) {
        // 噴氣移動
        if (keys['ArrowUp'])    this.vy -= this.accel;
        if (keys['ArrowDown'])  this.vy += this.accel;
        if (keys['ArrowLeft'])  this.vx -= this.accel;
        if (keys['ArrowRight']) this.vx += this.accel;

        // 穩定器邏輯
        if (this.dampener) {
            this.vx *= this.friction;
            this.vy *= this.friction;
        }

        this.x += this.vx;
        this.y += this.vy;
    }
}

<script>
const canvas = document.getElementById('gameCanvas');
const ctx = canvas.getContext('2d');
const dampenerDisplay = document.getElementById('dampener-status');

// 初始化畫布大小
function resize() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
}
window.addEventListener('resize', resize);
resize();

// 遊戲邏輯變數
let keys = {};
let player = {
    x: canvas.width / 2,
    y: canvas.height / 2,
    vx: 0,
    vy: 0,
    accel: 0.15,
    friction: 0.95, // 穩定器煞車力道
    isDampenerOn: true,
    radius: 15
};

// 監聽鍵盤
window.addEventListener('keydown', e => {
    keys[e.key] = true;
    if (e.key.toLowerCase() === 't') {
        player.isDampenerOn = !player.isDampenerOn;
        updateUI();
    }
});
window.addEventListener('keyup', e => keys[e.key] = false);

function updateUI() {
    dampenerDisplay.innerText = player.isDampenerOn ? "ON" : "OFF";
    dampenerDisplay.className = "status " + (player.isDampenerOn ? "on" : "off");
}

function update() {
    // 噴氣加速
    if (keys['ArrowUp'])    player.vy -= player.accel;
    if (keys['ArrowDown'])  player.vy += player.accel;
    if (keys['ArrowLeft'])  player.vx -= player.accel;
    if (keys['ArrowRight']) player.vx += player.accel;

    // 穩定器邏輯 (自動減速)
    if (player.isDampenerOn) {
        // 如果沒按按鍵，就施加摩擦力
        if (!keys['ArrowUp'] && !keys['ArrowDown']) player.vy *= player.friction;
        if (!keys['ArrowLeft'] && !keys['ArrowRight']) player.vx *= player.friction;
    }

    // 更新位置
    player.x += player.vx;
    player.y += player.vy;

    // 邊界反彈 (簡單處理)
    if (player.x < 0 || player.x > canvas.width) player.vx *= -0.5;
    if (player.y < 0 || player.y > canvas.height) player.vy *= -0.5;
}

function draw() {
    // 清除畫面
    ctx.fillStyle = '#050505';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // 繪製太空人 (暫時用圓圈代替)
    ctx.save();
    ctx.translate(player.x, player.y);
    
    // 畫出太空人主體
    ctx.beginPath();
    ctx.arc(0, 0, player.radius, 0, Math.PI * 2);
    ctx.fillStyle = 'white';
    ctx.fill();
    
    // 畫一個簡單的面罩，代表方向
    ctx.fillStyle = '#333';
    ctx.fillRect(5, -5, 10, 10);
    
    ctx.restore();

    // 循環執行
    update();
    requestAnimationFrame(draw);
}

// 啟動遊戲
draw();
