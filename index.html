// 核心邏輯預覽
class Astronaut {
    constructor() {
        this.x = 400;
        this.y = 300;
        this.vx = 0; // 水平速度
        this.vy = 0; // 垂直速度
        this.accel = 0.2; // 噴氣加速度
        this.friction = 0.96; // 穩定器摩擦力
        this.dampener = true; // 穩定器開關
    }

    update(keys) {
        // 噴氣移動
        if (keys['ArrowUp'])    this.vy -= this.accel;
        if (keys['ArrowDown'])  this.vy += this.accel;
        if (keys['ArrowLeft'])  this.vx -= this.accel;
        if (keys['ArrowRight']) this.vx += this.accel;

        // 穩定器邏輯
        if (this.dampener) {
            this.vx *= this.friction;
            this.vy *= this.friction;
        }

        this.x += this.vx;
        this.y += this.vy;
    }
}
